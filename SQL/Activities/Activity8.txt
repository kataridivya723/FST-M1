--Write an SQL statement to find the highest purchase amount ordered by the each customer with their ID and highest purchase amount.
SELECT customer_id, MAX(PURCHASE_AMOUNT) as "Highest Amount" from ORDERS Group By customer_id;
--Write an SQL statement to find the highest purchase amount on '2012-08-17' for each salesman with 
--their ID.
SELECT SALESMAN_ID, MAX(PURCHASE_AMOUNT)AS "MAX Amount" FROM ORDERS WHERE order_date=TO_DATE('2012-08-17','YYYY-MM-DD') Group By SALESMAN_ID,order_date;

--Write an SQL statement to find the highest purchase amount with their ID and order date, 
--for only those customers who have a higher purchase amount within the list [2030, 3450, 5760, 6000].
SELECT CUSTOMER_ID,ORDER_DATE,  MAX(PURCHASE_AMOUNT) AS "MAX AMOUNT" FROM ORDERS
Group By customer_id,order_date
having MAX(PURCHASE_AMOUNT) IN (2030, 3450, 5760, 6000);
